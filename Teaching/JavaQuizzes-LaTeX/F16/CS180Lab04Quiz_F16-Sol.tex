public class Album {
	public Artist artist;
	public String albumTitle;

	public Album(String title, Artist artist) {
		this.albumTitle = title;
		this.artist = artist;
	}
	public String getTitle() {
		return this.albumTitle;
	}
	public Artist getArtist() {
		return this.artist;
	}
}

	}
	public String getName() {
